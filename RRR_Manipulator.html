function btn_Forward_Callback(hObject, eventdata, handles)
% hObject    handle to btn_Forward (see GCBO)
% eventdata  reserved - to be defined in a future version of MATLAB
% handles    structure with handles and user data (see GUIDATA)

Robot = SerialLink(L);

% Perform forward kinematics to get the end effector position
T = Robot.fkine([Th_1, Th_2, Th_3]);

    function btn_Inverse_Callback(hObject, eventdata, handles)
% hObject    handle to btn_Inverse (see GCBO)
% eventdata  reserved - to be defined in a future version of MATLAB
% handles    structure with handles and user data (see GUIDATA)

PX = str2double(get(handles.Pos_X, 'String'));
PY = str2double(get(handles.Pos_Y, 'String'));
PZ = str2double(get(handles.Pos_Z, 'String'));

% Check if the entered positions are valid numbers
if isnan(PX) || isnan(PY) || isnan(PZ)
    errordlg('Invalid position values. Please enter numeric values for X, Y, and Z.', 'Input Error');
    return;
end

% Define link lengths
L_1 = 20;
L_2 = 50;
L_3 = 40;

% Create the robot links
L(1) = Link([0 L_1 0 pi/2]);
L(2) = Link([0 0 L_2 0]);
L(3) = Link([0 0 L_3 0]);

% Create the serial link robot
Robot = SerialLink(L);
Robot.name = 'RRR Robot';

% Create the transformation matrix for the desired end effector position
T = transl(PX, PY, PZ);

% Solve for the inverse kinematics with a mask [1 1 1 0 0 0] and a better initial guess
try
    % Initial guess for the joint angles
    initial_guess = [0, 0, 0];
    J = Robot.ikine(T, 'mask', [1 1 1 0 0 0], 'q0', initial_guess);

    % Convert from radians to degrees
    J = J * 180 / pi;

    % Update the joint angle displays
    set(handles.Theta_1, 'String', num2str(floor(J(1))));
    set(handles.Theta_2, 'String', num2str(floor(J(2))));
    set(handles.Theta_3, 'String', num2str(floor(J(3))));

catch ME
    % Display an error dialog if inverse kinematics fails
    errordlg(['Inverse kinematics failed: ', ME.message], 'IK Error');
end



% --- Executes on button press in btn_Simulation.
function btn_Simulation_Callback(hObject, eventdata, handles)
% hObject    handle to btn_Simulation (see GCBO)
% eventdata  reserved - to be defined in a future version of MATLAB
% handles    structure with handles and user data (see GUIDATA)

% Get the angles from the GUI inputs and convert to radians
Th_1 = str2double(get(handles.Theta_1, 'String')) * pi / 180;
Th_2 = str2double(get(handles.Theta_2, 'String')) * pi / 180;
Th_3 = str2double(get(handles.Theta_3, 'String')) * pi / 180;

% Define link lengths
L_1 = 20;
L_2 = 50;
L_3 = 40;

% Create the robot links
L(1) = Link([0 L_1 0 pi/2]);
L(2) = Link([0 0 L_2 0]);
L(3) = Link([0 0 L_3 0]);

% Create the serial link robot
Robot = SerialLink(L);
Robot.name = 'RRR Robot';

% Perform forward kinematics to get the end effector position
T = Robot.fkine([Th_1, Th_2, Th_3]);

% Update the position display in the GUI
set(handles.Pos_X, 'String', num2str(floor(T.t(1))));
set(handles.Pos_Y, 'String', num2str(floor(T.t(2))));
set(handles.Pos_Z, 'String', num2str(floor(T.t(3))));

% Define the initial and final joint configurations
initialConfig = [0, 0, 0]; % Change this if needed
finalConfig = [Th_1, Th_2, Th_3];

% Number of intermediate steps for the slow-motion effect
numSteps = 100;

% Generate intermediate configurations
interpConfigs = interp1([1, numSteps], [initialConfig; finalConfig], 1:numSteps);

% Plot the robot configuration in slow motion
axes(handles.axes1);
for i = 1:numSteps
    Robot.plot(interpConfigs(i, :), 'workspace', [-100 100 -100 100 -100 100]);
    pause(0.05); % Pause to create the slow-motion effect
end
